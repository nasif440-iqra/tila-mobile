**Step 1: Create `src/hooks/useThemePreference.ts`**

Hook that reads and writes theme preference from user_profile.theme_mode:
```typescript
import { useState, useEffect, useCallback } from 'react';
import { useDatabase } from '../db/provider';
import type { ThemeMode } from '../design/theme';

export function useThemePreference() {
  const db = useDatabase();
  const [themeMode, setThemeMode] = useState<ThemeMode>('system');
  const [loaded, setLoaded] = useState(false);

  useEffect(() => {
    db.getFirstAsync<{ theme_mode: string }>(
      "SELECT theme_mode FROM user_profile WHERE id = 1"
    ).then((row) => {
      if (row?.theme_mode) {
        setThemeMode(row.theme_mode as ThemeMode);
      }
      setLoaded(true);
    }).catch(() => {
      setLoaded(true); // Default to 'system' on error
    });
  }, [db]);

  const updateThemeMode = useCallback(async (mode: ThemeMode) => {
    setThemeMode(mode);
    await db.runAsync(
      "UPDATE user_profile SET theme_mode = ?, updated_at = datetime('now') WHERE id = 1",
      mode
    );
  }, [db]);

  return { themeMode, updateThemeMode, loaded };
}
```

**Step 3: Update `app/_layout.tsx`** ‚Äî Remove forced light mode, use dynamic preference

The key change: replace the hardcoded `useState<ThemeMode>("light")` with the `useThemePreference` hook. However, `useThemePreference` uses `useDatabase()` which requires `DatabaseProvider` ‚Äî but `ThemeContext` wraps DatabaseProvider. This creates a chicken-and-egg problem.

Solution: Split into two stages:
1. Pre-DB: use system scheme directly for initial render (before DB is ready)
2. Post-DB: use `useThemePreference` inside a child component

Create a `ThemeWrapper` inner component that lives inside `DatabaseProvider`:
```typescript
function ThemeWrapper({ children }: { children: ReactNode }) {
  const systemScheme = useColorScheme();
  const { themeMode, loaded } = useThemePreference();
  
  // Use system scheme until preference is loaded
  const effectiveMode = loaded ? themeMode : 'system';
  const { colors, mode } = resolveColors(effectiveMode, systemScheme);
  
  return (
    <ThemeContext.Provider value={{ colors, mode }}>
      {children}
    </ThemeContext.Provider>
  );
}
```

**Step 1: Add tiered content helper**
Create a function that returns appropriate content based on days since last practice:

```typescript
interface ReturnContent {
  greeting: string;
  hadith: string;
  attribution: string;
  encouragement: string;
  buttonText: string;
}

function getReturnContent(daysSince: number, streakBroke: boolean, longestWird: number): ReturnContent {
  if (daysSince <= 1) {
    // Yesterday or today ‚Äî light "welcome back"
    return {
      greeting: 'Welcome back',
      hadith: '"The most beloved of deeds to Allah are those that are most consistent, even if they are small."',
      attribution: '‚Äî Prophet Muhammad (peace be upon him)',
      encouragement: 'Every return is a step forward. Pick up right where you left off.',
      buttonText: 'Continue',
    };
  } else if (daysSince <= 7) {
    // 2-7 days ‚Äî gentle "we missed you"
    return {
      greeting: 'We missed you',
      hadith: '"Whoever travels a path in search of knowledge, Allah will make easy for him a path to paradise."',
      attribution: '‚Äî Prophet Muhammad (peace be upon him)',
      encouragement: streakBroke
        ? `Your ${longestWird}-day streak shows real dedication. Let's build another one.`
        : `You've been away for ${daysSince} days. Your progress is right where you left it.`,
      buttonText: streakBroke ? 'Start fresh' : 'Continue learning',
    };
  } else {
    // 14+ days ‚Äî warm "it's never too late"
    return {
      greeting: 'It\'s never too late',
      hadith: '"The best time to plant a tree was twenty years ago. The second best time is now."',
      attribution: '‚Äî Islamic proverb',
      encouragement: streakBroke
        ? `You reached a ${longestWird}-day streak before ‚Äî that strength is still in you. One lesson is all it takes.`
        : 'Every great journey has pauses. Your learning is waiting for you, exactly where you left off.',
      buttonText: 'Begin again',
    };
  }
}
```
