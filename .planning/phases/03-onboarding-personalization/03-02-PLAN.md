  useEffect(() => {
    hapticSelection();
    playSacredMoment();
  }, []);

  return (
    <OnboardingStepLayout
      variant="splash"
      fadeInDuration={800}
      footer={
        revealComplete ? (
          <Animated.View entering={FadeIn.duration(400)} style={{ zIndex: 1 }}>
            <Button title="Continue" onPress={onNext} />
          </Animated.View>
        ) : undefined
      }

      <PhraseReveal
        words={BISMILLAH_WORDS}
        layout="vertical"
        onComplete={() => setRevealComplete(true)}
        accessibilityLabel="Bismillah ir-Rahman ir-Raheem. In the name of God, the Most Gracious, the Most Merciful."
      />
    </OnboardingStepLayout>
  );
}
