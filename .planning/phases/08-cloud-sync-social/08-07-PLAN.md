A mock that simulates Supabase client behavior for testing:
```typescript
import { vi } from 'vitest';

interface MockRow {
  [key: string]: unknown;
}

export function createMockSupabase(initialData: Record<string, MockRow[]> = {}) {
  const data: Record<string, MockRow[]> = { ...initialData };
  
  const from = (table: string) => ({
    select: (columns: string = '*') => ({
      eq: (column: string, value: unknown) => ({
        data: (data[table] ?? []).filter(row => row[column] === value),
        error: null,
      }),
    }),
    upsert: (rows: MockRow[], options?: { onConflict: string }) => {
      if (!data[table]) data[table] = [];
      for (const row of Array.isArray(rows) ? rows : [rows]) {
        const existing = data[table].findIndex(r => 
          r[options?.onConflict?.split(',')[1]?.trim() ?? 'id'] === row[options?.onConflict?.split(',')[1]?.trim() ?? 'id']
        );
        if (existing >= 0) data[table][existing] = { ...data[table][existing], ...row };
        else data[table].push(row);
      }
      return { data: rows, error: null };
    },
    insert: (rows: MockRow[]) => {
      if (!data[table]) data[table] = [];
      data[table].push(...(Array.isArray(rows) ? rows : [rows]));
      return { data: rows, error: null };
    },
    delete: () => ({
      eq: (column: string, value: unknown) => {
        if (data[table]) {
          data[table] = data[table].filter(row => row[column] !== value);
        }
        return { error: null };
      },
    }),
  });
  
  const auth = {
    onAuthStateChange: vi.fn((callback: Function) => {
      return { data: { subscription: { unsubscribe: vi.fn() } } };
    }),
    signInWithIdToken: vi.fn().mockResolvedValue({ data: { user: { id: 'test-user-id' }, session: {} }, error: null }),
    signInWithPassword: vi.fn().mockResolvedValue({ data: { user: { id: 'test-user-id' }, session: {} }, error: null }),
    signUp: vi.fn().mockResolvedValue({ data: { user: { id: 'test-user-id' }, session: {} }, error: null }),
    signOut: vi.fn().mockResolvedValue({ error: null }),
    updateUser: vi.fn().mockResolvedValue({ data: {}, error: null }),
    getSession: vi.fn().mockResolvedValue({ data: { session: null }, error: null }),
  };
  
  return { from, auth, _data: data };
}

export type MockSupabaseClient = ReturnType<typeof createMockSupabase>;
```

```typescript
import { vi } from 'vitest';

interface MockRow {
  [key: string]: unknown;
}

export function createMockDb(initialTables: Record<string, MockRow[]> = {}) {
  const tables: Record<string, MockRow[]> = { ...initialTables };
  
  return {
    getAllAsync: vi.fn(async (sql: string) => {
      // Parse table name from "SELECT * FROM tablename"
      const match = sql.match(/FROM\s+(\w+)/i);
      const table = match?.[1] ?? '';
      return tables[table] ?? [];
    }),
    getFirstAsync: vi.fn(async (sql: string) => {
      const match = sql.match(/FROM\s+(\w+)/i);
      const table = match?.[1] ?? '';
      return (tables[table] ?? [])[0] ?? null;
    }),
    runAsync: vi.fn(async () => ({ changes: 1 })),
    execAsync: vi.fn(async () => {}),
    withExclusiveTransactionAsync: vi.fn(async (fn: Function) => {
      // Execute the transaction function with the mock db itself
      await fn({
        getAllAsync: vi.fn(async () => []),
        getFirstAsync: vi.fn(async () => null),
        runAsync: vi.fn(async () => ({ changes: 1 })),
        execAsync: vi.fn(async () => {}),
      });
    }),
    _tables: tables,
  };
}

export type MockDb = ReturnType<typeof createMockDb>;
```
