**Step 1: Create `src/state/provider.tsx`** — AppStateProvider (per RET-02)

This provider wraps the existing hooks into a single canonical state source:
```typescript
import { createContext, useCallback, useMemo, type ReactNode } from 'react';
import { useProgress } from '../hooks/useProgress';
import { useHabit } from '../hooks/useHabit';
import type { AppStateContextValue } from './types';

export const AppStateContext = createContext<AppStateContextValue | null>(null);

export function AppStateProvider({ children }: { children: ReactNode }) {
  const progress = useProgress();
  const { habit, refresh: refreshHabit } = useHabit();
  
  // SubscriptionState comes from SubscriptionProvider above us in the tree
  // We don't duplicate it — consumers use useSubscription() directly
  
  const refreshAll = useCallback(async () => {
    await Promise.all([
      progress.refresh(),
      refreshHabit(),
    ]);
  }, [progress.refresh, refreshHabit]);
  
  const value = useMemo<AppStateContextValue>(() => ({
    progress: progress.loading ? null : {
      completedLessons: progress.completedLessons,
      lessonAttempts: progress.lessonAttempts,
      mastery: progress.mastery,
      profile: progress.profile,
    } as any, // Will be properly typed from ProgressState
    habit,
    subscription: null, // Accessed via useSubscription() directly
    loading: progress.loading,
    refreshAll,
  }), [progress, habit, refreshAll]);
  
  return (
    <AppStateContext.Provider value={value}>
      {children}
    </AppStateContext.Provider>
  );
}
```
