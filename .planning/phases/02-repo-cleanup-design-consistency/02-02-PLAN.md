    Create `src/design/CrescentIcon.tsx`:
    ```typescript
    import React from 'react';
    import Svg, { Circle } from 'react-native-svg';

    interface CrescentIconProps {
      size?: number;
      color?: string;
    }

    /**
     * Crescent moon icon matching the BrandedLogo style.
     * Two overlapping circles: a filled main circle and a transparent cutout.
     *
     * Proportions derived from BrandedLogo.tsx:
     *   Main circle: cx=200 cy=160 r=52
     *   Cutout circle: cx=218 cy=146 r=42
     * Normalized to a 24x24 viewBox for icon usage.
     */
    export function CrescentIcon({ size = 16, color = '#C4A464' }: CrescentIconProps) {
      // Normalize BrandedLogo coordinates (200,160 with r=52) to fit 24x24 viewBox
      // Original bounding box: x=[148,252], y=[108,212] -> 104x104 range
      // Scale factor: 24/104 = 0.2308
      // Offset to center: shift x by -148, y by -108, then scale
      return (
        <Svg width={size} height={size} viewBox="0 0 24 24">
          <Circle cx={12} cy={12} r={12} fill={color} />
          <Circle cx={16.15} cy={8.77} r={9.69} fill="transparent" />
        </Svg>
      );
    }
    ```

    The proportions maintain the same visual ratio as BrandedLogo:
    - Main circle centered at (12, 12) with r=12 (fills the viewBox)
    - Cutout circle offset by (+4.15, -3.23) with r=9.69 (proportional to 42/52 * 12)
    - Cutout uses `fill="transparent"` so it works over any background (per research pitfall 4)

    The default color `#C4A464` matches `colors.accent` (gold). Callers can pass `color={colors.accent}` explicitly via useColors() for theme consistency.
